async function initUser() {
    const user = tg?.initDataUnsafe?.user;
    if (!user) {
        currentUser = { id: 123456, name: "Гость", total_score: 0 };
        return;
    }
    
    currentUser = {
        id: user.id,
        name: user.first_name || "Гость",
        avatar: user.photo_url || null
    };
    
    try {
        const { data, error } = await supabaseClient
            .from('users')
            .select('*')
            .eq('telegram_id', String(user.id))
            .single();
        
        if (!data) {
            await supabaseClient.from('users').insert({
                telegram_id: String(user.id),
                name: currentUser.name,
                avatar: currentUser.avatar,
                total_score: 0
            });
        } else {
            currentUser.total_score = data.total_score || 0;
        }
    } catch(e) {
        console.log("Supabase error:", e);
    }
}

async function saveResult(score, maxScore, lessonId) {
    if (!currentUser || currentUser.id === 123456) return;
    
    try {
        await supabaseClient.from('results').insert({
            telegram_id: String(currentUser.id),
            topic_id: lessonId,
            score: score,
            max_score: maxScore
        });
        
        const { data } = await supabaseClient
            .from('results')
            .select('score')
            .eq('telegram_id', String(currentUser.id));
        
        let total = 0;
        if (data) data.forEach(r => total += r.score);
        
        await supabaseClient
            .from('users')
            .update({ total_score: total })
            .eq('telegram_id', String(currentUser.id));
        
        currentUser.total_score = total;
    } catch(e) {
        console.log("Save error:", e);
    }
}

async function loadRating() {
    try {
        const { data } = await supabaseClient
            .from('users')
            .select('name, total_score')
            .order('total_score', { ascending: false })
            .limit(20);
        
        const container = document.getElementById('ratingList');
        if (!data || data.length === 0) {
            container.innerHTML = '<div class="stat-card">⭐ Пока никого нет</div>';
            return;
        }
        
        container.innerHTML = data.map((user, i) => `
            <div class="rank-item">
                <div class="rank-avatar">${i === 0 ? '🏆' : (i === 1 ? '🥈' : (i === 2 ? '🥉' : '📖'))}</div>
                <div style="flex:1"><strong>${user.name || "Аноним"}</strong></div>
                <div style="font-weight:bold;color:#d48c9c">${user.total_score || 0} баллов</div>
            </div>
        `).join('');
    } catch(e) {
        document.getElementById('ratingList').innerHTML = '<div class="stat-card">⭐ Рейтинг временно недоступен</div>';
    }
}

function switchTab(tab) {
    currentTab = tab;
    
    document.querySelectorAll('.tab').forEach(t => t.classList.remove('active'));
    document.querySelector(`.tab[data-tab="${tab}"]`).classList.add('active');
    
    document.getElementById('subjectsList').style.display = 'none';
    document.getElementById('statsContent').style.display = 'none';
    document.getElementById('ratingList').style.display = 'none';
    document.getElementById('profileContent').style.display = 'none';
    
    if (tab === 'subjects') {
        document.getElementById('subjectsList').style.display = 'block';
        renderMain();
    }
    if (tab === 'stats') {
        document.getElementById('statsContent').style.display = 'block';
        renderStats();
    }
    if (tab === 'rating') {
        document.getElementById('ratingList').style.display = 'block';
        loadRating();
    }
    if (tab === 'profile') {
        document.getElementById('profileContent').style.display = 'block';
        renderProfile();
    }
}

function startQuiz(lesson) {
    currentLesson = lesson;
    currentQuestionIndex = 0;
    currentScore = 0;
    answered = false;
    
    document.getElementById('quizTitle').innerText = lesson.title;
    document.getElementById('mainScreen').style.display = 'none';
    document.getElementById('quizScreen').style.display = 'block';
    document.getElementById('resultScreen').style.display = 'none';
    renderQuestion();
}

function renderQuestion() {
    const q = currentLesson.questions[currentQuestionIndex];
    if (!q) return finishQuiz();
    
    document.getElementById('questionText').innerHTML = q.text;
    const percent = (currentQuestionIndex / currentLesson.questions.length) * 100;
    document.getElementById('progressFill').style.width = percent + "%";
    
    const box = document.getElementById('optionsBox');
    box.innerHTML = '';
    selectedOption = null;
    answered = false;
    
    const btn = document.getElementById('actionBtn');
    btn.disabled = true;
    btn.classList.remove('active');
    btn.innerText = "Ответить";
    
    q.options.forEach((opt, i) => {
        const div = document.createElement('div');
        div.className = 'option';
        div.innerText = opt;
        div.onclick = () => {
            if (answered) return;
            selectedOption = i;
            document.querySelectorAll('.option').forEach(o => o.classList.remove('selected'));
            div.classList.add('selected');
            btn.disabled = false;
            btn.classList.add('active');
        };
        box.appendChild(div);
    });
}

function handleAnswer() {
    if (selectedOption === null) return;
    
    const q = currentLesson.questions[currentQuestionIndex];
    const btn = document.getElementById('actionBtn');
    
    if (!answered) {
        answered = true;
        document.querySelectorAll('.option').forEach((el, i) => {
            if (i === q.correct) el.classList.add('correct');
            else if (i === selectedOption) el.classList.add('wrong');
        });
        if (selectedOption === q.correct) currentScore++;
        btn.innerText = "Дальше →";
    } else {
        currentQuestionIndex++;
        if (currentQuestionIndex < currentLesson.questions.length) renderQuestion();
        else finishQuiz();
    }
}

async function finishQuiz() {
    await saveResult(currentScore, currentLesson.questions.length, currentLesson.id);
    
    document.getElementById('quizScreen').style.display = 'none';
    document.getElementById('resultScreen').style.display = 'block';
    const percent = Math.round((currentScore / currentLesson.questions.length) * 100);
    document.getElementById('resultScore').innerHTML = `${currentScore}/${currentLesson.questions.length}<div style="font-size:16px;">${percent}%</div>`;
}

function backToMain() {
    document.getElementById('mainScreen').style.display = 'block';
    document.getElementById('quizScreen').style.display = 'none';
    document.getElementById('resultScreen').style.display = 'none';
    if (currentTab === 'stats') renderStats();
    if (currentTab === 'rating') loadRating();
    if (currentTab === 'profile') renderProfile();
    switchTab(currentTab);
}

function retryQuiz() {
    currentScore = 0;
    currentQuestionIndex = 0;
    answered = false;
    document.getElementById('resultScreen').style.display = 'none';
    document.getElementById('quizScreen').style.display = 'block';
    renderQuestion();
}

document.querySelectorAll('.tab').forEach(tab => {
    tab.addEventListener('click', () => switchTab(tab.dataset.tab));
});
